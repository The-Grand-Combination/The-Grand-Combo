	#Description: Norway asks for independence from Sweden - accept or refuse?
	country_event = {

		id = 49505

		title = "EVTNAME49505"
		desc = "EVTDESC49505"

		trigger = {
			war = no
			tag = SWE
			any_state = {
				any_owned_province = { is_core = NOR }

			}
			average_consciousness = 6
			year = 1880
			revolution_n_counterrevolution = 1
			is_possible_vassal = NOR
			NOT = {
				has_country_flag = norway_asks_for_indepenedence
				exists = NOR
			}
		}

		mean_time_to_happen = {
			days = 1
			modifier = {
				factor = 0.5
				press_rights = free_press
			}
			modifier = {
				factor = 0.75
				vote_franschise = universal_voting
			}
			modifier = {
				factor = 0.85
				vote_franschise = universal_weighted_voting
			}
			modifier = {
				factor = 0.95
				vote_franschise = wealth_voting
				}
			modifier = {
				factor = 1.25
				press_rights = censored_press
			}
			modifier = {
				factor = 1.5
				press_rights = state_press
			}
		}

		immediate = { random_owned = { limit = { owner = { accepted_culture = norwegian } } owner = { remove_accepted_culture = norwegian } } }

		#Accept
		option = {
			name = "EVTOPTA49505"

			any_owned = {
				limit = { is_core = NOR }
				remove_core = SWE
			}
			release = NOR
			any_pop = {
				consciousness = -4
			}
			NOR = {
				government = hms_government
				any_pop = {
					militancy = -4
				}
				relation = {
					who = THIS
					value = 100
				}
			}
			set_country_flag = norway_asks_for_indepenedence

			ai_chance = { factor = 90 }
		}

		#Refuse
		option = {
			name = "EVTOPTB49505"

			any_pop = {
				limit = { culture = norwegian }
				consciousness = 4
				militancy = 3
			}
			any_country = {
				limit = {
					is_greater_power = yes
					continent = europe
				}
				relation = {
					who = THIS
					value = -20
				}
			}
			set_country_flag = norway_asks_for_indepenedence

			ai_chance = { factor = 10 }
		}
	}
