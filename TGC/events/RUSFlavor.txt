				has_country_flag = refused_alaska
				any_owned_province = { is_core = RUS }
				truce_with = RUS
			}
			relation = { who = RUS value = 0 }
			war = no
			nationalism_n_imperialism = 1
		}

		mean_time_to_happen = {
			months = 6
			modifier = {
				factor = 0.5
				tag = USA
			}
			modifier = {
				factor = 0.8
				alliance_with = RUS
			}
			modifier = {
				factor = 0.8
				relation = { who = RUS value = 100 }
			}
			modifier = {
				factor = 1.2
				NOT = { relation = { who = RUS value = 50 } }
			}
			modifier = {
				factor = 0.1
				year = 1870
			}
		}

		immediate = {
			set_global_flag = alaska_purchase_process_global
		}

		# Make an offer
		option = {
			name = "EVTOPTA32509"

			random_owned = {
				limit = { owner = { ai = no } }
				owner = { treasury = -100000 }
			}
			random_owned = {
				limit = { owner = { ai = yes NOT = { check_variable = { which = owed_cash_money value = 1 } } } }

			random_owned = {
				limit = { owner = { ai = yes check_variable = { which = owed_cash_money value = 9 } } }

			}

			RUS = { country_event = 32510 }

			ai_chance = { factor = 95 }
		}

		# Not interested
		option = {
			name = "EVTOPTB32509"

			clr_country_flag = alaska_purchase_interest
			set_country_flag = refused_alaska

				factor = 5
				modifier = {
					factor = 0
					OR = {
						tag = USA
						tag = CAN
					}
				}
			}
		}
	}
