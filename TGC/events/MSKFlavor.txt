#The Treaty of Managua

	#Description: Triggered by the UK if they still have the Moskito Kingdom as a vassal and want to sell it.
	country_event = {
		id = 99700
		picture = "administration"
		title = "EVTNAME99700"
		desc = "EVTDESC99700"

		major = yes
		is_triggered_only = yes

		#Accept and get MSK as a vassal
		option = {
			name = "EVTOPTA99700"

			set_global_flag = treaty_of_managua_success
			badboy = -2
			random_owned = { limit = { owner = { ai = no } } owner = { treasury = -1000 } }
			FROM = {
				badboy = -2
				prestige = 5
				release = MSK
				release_vassal = MSK
				diplomatic_influence = {
					who = MSK
					value = -300
				}
				leave_alliance = MSK
				end_military_access = MSK
			}
			relation = {
				who = FROM
				value = 100
			}
			random_owned = { limit = { owner = { is_greater_power = no } }
				FROM = {
					diplomatic_influence = {
						who = THIS
						value = 100
					}
				}
			}
			any_pop = {
				limit = { is_primary_culture = yes }
				militancy = -1
				consciousness = -1
			}

			3479 = {
				remove_core = MSK
			}

			inherit = MSK
			release_vassal = MSK
			MSK = { government = colonial_company treasury = 1000 }
			random_owned = { limit = { owner = { is_greater_power = yes } }
				owner = {
					diplomatic_influence = {
						who = MSK
						value = 400
					}
				}
			}

			ai_chance = { factor = 1 }
		}

		#Refuse and get a CB to take it by force
		option = {
			name = "EVTOPTB99700"

			prestige = 5
			badboy = 2
			any_pop = {
				dominant_issue = {
					factor = 0.02
					value = jingoism
				}
			}
			casus_belli = {
				target = FROM
				type = release_puppet
				months = 30
			}
			relation = {
				who = FROM
				value = -200
			}
			leave_alliance = FROM
			end_military_access = FROM
			random_owned = { limit = { owner = { is_greater_power = no } }
				FROM = {
					diplomatic_influence = {
						who = THIS
						value = -400
					}
				}
			}
			any_pop = {
				limit = { is_primary_culture = yes }
				militancy = -2
				consciousness = -2
			}

			ai_chance = { factor = 0 }
		}
	}

	#Description: MSK can occupy San Juan del Norte or choose not to
	country_event = {

		title = "EVTNAME99701"
		desc = "EVTDESC99701"

		trigger = {

			year = 1848
		}

			months = 3

		# Take it
		option = {
