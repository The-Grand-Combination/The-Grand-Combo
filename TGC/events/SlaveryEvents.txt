## Slave Trade ##

	#Description: The Slave Trade - buy slaves?
	country_event = {
		id = 184000
		title = "EVTNAME184000"
		desc = "EVTDESC184000"
		picture = "slaves_traded"

		trigger = {
			OR = {
				has_country_flag = slave_trade_leader
				has_country_flag = slave_trader
			}
			NOT = { has_country_modifier = slave_trade_cooldown_country }
			money = 21000
		}

		mean_time_to_happen = {
			days = 7
		}

		# Buy more slaves
		option = {
			name = "EVTOPTA184000"

			any_pop = {
				limit = { type = slaves }
				reduce_pop = 1.04
			}
			money = -10000
			add_country_modifier = { name = slave_trade_cooldown_country duration = 1825 } # 5 Years

			ai_chance = {
				factor = 70
				modifier = {
					factor = 0
					NOT = { money = 200000 }
				}
			}
		}

		# Don't buy any more slaves
		option = {
			name = "EVTOPTB184000"

			add_country_modifier = { name = slave_trade_cooldown_country duration = 365 }

			ai_chance = { factor = 30 }
		}
	}

	#Description: The Obsolescence of slavery - keep slavery or industrialize?
	country_event = {
		id = 181000
		title = "EVTNAME181000"
		desc = "EVTDESC181000"
		major = yes

		trigger = {
			slavery = yes_slavery
			civilized = yes
			year = 1890
			OR = {
				average_consciousness = 4
				mass_politics = 1
			}
			NOT = { has_country_flag = slavery_still_stands }
		}

		mean_time_to_happen = {
			months = 36
			modifier = {
				factor = 0.9
				average_consciousness = 6
			}
			modifier = {
				factor = 0.5
				synthetic_polymers = 1
			}
			modifier = {
				factor = 2
				NOT = { tag = CSA }
			}
		}

		fire_only_once = yes

		# Abolish slavery
		option = {
			name = "EVTOPTA181000"

			political_reform = no_slavery
			add_country_modifier = {
				name = onward_to_progress
				duration = 3650
			}
			clr_country_flag = the_slavery_debate
			remove_country_modifier = the_slavery_debate_modifier
			remove_country_modifier = slavery_timer
			set_country_flag = slavery_ended_event
			nationalvalue = nv_liberty
			add_country_modifier = {
				name = national_confusion
				duration = 3650
			}
			prestige = 20
			badboy = -6
			any_pop = {
				militancy = -3
				consciousness = -1
			}

			ai_chance = {
				factor = 70
				modifier = {
					factor = 0.2
					NOT = { badboy = 0.1 }
				}
			}
		}

		# Keep slavery
		option = {
			name = "EVTOPTB181000"

			remove_country_modifier = the_slavery_debate_modifier
			set_country_flag = slavery_still_stands
			badboy = 3
			any_pop = {
				militancy = 2
				consciousness = 2
			}
			add_country_modifier = { name = mass_labour duration = -1 }

			ai_chance = {
				factor = 30
				modifier = {
					factor = 0.5
					average_militancy = 7
				}
			}
		}
	}
