#Ethnic Flavour

    desc = "EVTDESC2000_NEW"

			OR = {

				}
			}
			is_colonial = no
		}
		citizenship_policy = full_citizenship
		civilized = yes

    mean_time_to_happen = {
		months = 250
		modifier = {

			factor = 0.9
			average_consciousness = 4
		}
		modifier = {
			factor = 1.5
			average_militancy = 4
		}
    }

    option = {
		name = "EVTOPTA2000"
		prestige = 2
    }

    option = {
		name = "EVTOPTB2000"
		random_owned = {
			limit = {
				OR = {
					continent = north_america
