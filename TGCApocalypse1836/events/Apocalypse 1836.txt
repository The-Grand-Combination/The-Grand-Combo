#A mysterious plague, outbreak
country_event = {
	id = 123456001
	title = "EVTNAME123456001"
	desc = "EVTDESC123456001"
	picture = disease

	fire_only_once = yes
	
	trigger = {
		ai = no
		NOT = {
			has_country_flag = zombie_outbreak
			has_country_flag = zombie_outbreak_defeat
			has_country_flag = zombie_outbreak_victory
		}
	}

	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		set_global_flag = zombie_mod
	}

	option = {
		name = "EVTOPTA123456001"
		random_country = {
			limit = {
				civilized = yes
				number_of_states = 3
			}
			country_event = 123456002
		}
	}
}

#Country to Province
country_event = {
	id = 123456002
	title = "EVTNAME123456002"
	desc = "EVTDESC123456002"
	picture = disease

	trigger = {
		any_neighbor_country = {
			has_country_flag = zombie_outbreak
		}
		NOT = {
			has_country_flag = zombie_outbreak
			has_country_flag = zombie_outbreak_defeat
			has_country_flag = zombie_outbreak_victory
		}
	}

	mean_time_to_happen = {
		months = 2
		modifier = { factor = 9 has_country_modifier = ap_lock_down }
	}

	option = {
		name = "EVTOPTA123456002"

		random_province = {
			limit = {
				is_capital = no
				is_colonial = no
				is_overseas = no
			}
			militancy = 8
			add_province_modifier = { name = zombie_outbreak duration = 10000 }
			any_pop = { consciousness = -2 }
		}
	}
}
