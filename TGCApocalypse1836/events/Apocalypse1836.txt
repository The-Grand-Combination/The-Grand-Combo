#Province Victory
province_event = {
	id = 66014
	title = "EVTNAME66014"
	desc = "EVTDESC66014"
	picture = "prov_victory"

	trigger = {
		has_province_modifier = zombie_outbreak
		NOT = {
			controlled_by = REB
			country_units_in_province = REB
		}
		OR = {
			units_in_province = 1 
			invention = the_cure
		}
		
	}

	mean_time_to_happen = {
		months = 3
		modifier = {
			factor = 0.1
			AND = { invention = the_cure units_in_province = 1 }
		}
		modifier = {
			factor = 0.2
			units_in_province = 1 
		}
		modifier = {
			factor = 0.5
			invention = the_cure
		}
	}

	option = {
		name = "Excellent."
		remove_province_modifier = zombie_outbreak
		any_pop = {
			consciousness = 2
			militancy = -8
		}
	}
}

#Conquest of the wasteland
country_event = {
	id = 66036
	title = "EVTNAME66036"
	desc = "EVTDESC66036"
	picture = retake_the_wasteland

	trigger = {
		exists = yes
		has_country_flag = zombie_outbreak_victory
		any_owned_province = { has_empty_adjacent_province = yes }
		NOT = { has_global_flag = infection_mode }
	}

	is_triggered_only = yes

	mean_time_to_happen = {
		months = 80
		
		modifier = { factor = 0.5 has_country_modifier = ap_agressive_rebuild }
		modifier = { factor = 0.7 has_country_modifier = ap_temperace_rebuild }
		modifier = { factor = 0.9 NOT = { has_country_modifier = ap_lock_down } }

		modifier = { factor = 0.1 has_country_flag = scouts_activated }
	}

	option = {
		name = "EVTOPTA66036"

		prestige = 15
		money = -50000

		random_province = {
			limit = { has_empty_adjacent_province = yes }
			add_province_modifier = { name = ap_expedition duration = -1 }
			owner = { clr_country_flag = scouts_activated }
		}
	}

	option = {
		name = "EVTOPTB66036"
	}
}
