province_event = {                                   #Zombify province
	id = 123456100
	title = "EVTNAME123456025"
	desc = "EVTDESC123456025"

	trigger = {
		owner = {exists = yes }
		NOT = { owner = { tag = ZOM } }
		province_control_days = 730
		has_province_modifier = zombie_outbreak
		controlled_by = REB
		exists = ZOM
		ZOM = {
			has_country_modifier = accepts_new_provinces
		}
	}

	mean_time_to_happen = {
		months = 10
		modifier = {
			factor = 0.75
			any_neighbor_province = {
				has_province_modifier = zombie_outbreak
			}
		}
		modifier = {
			factor = 0.75
			has_empty_adjacent_province = yes
		}

		
	}

	option = {
		name = "Uh oh."
		any_pop = {
			reduce_pop = 0.4
			militancy = 8
			consciousness = -3
		}
		add_core = THIS
		owner = { prestige = -3 }
		life_rating = -10
		change_controller = ZOM
		secede_province = ZOM
		ZOM = {
			remove_country_modifier = do_not_annex
			add_country_modifier = {
				name = do_not_annex
				duration = 31
			}
		}
	}
}

country_event = { #note, this must be a country event, otherwise it might happen multiple times at once, which can cause a crash
	id = 123456101
	title = "EVTNAME123456101"
	desc = "EVTDESC123456101"

	trigger = {
		exists = yes
		NOT = { exists = ZOM }
		any_owned_province = {
			province_control_days = 730
			has_province_modifier = zombie_outbreak
			controlled_by = REB
		}
	}

	mean_time_to_happen = {
		months = 10
	}

	option = {
		name = "Uh oh."
		random_owned = {
			limit = {
				province_control_days = 730
				has_province_modifier = zombie_outbreak
				controlled_by = REB
			}
			any_pop = {
				reduce_pop = 0.4
				militancy = 8
				consciousness = -3
			}
			add_core = THIS
			owner = { prestige = -3 }
			life_rating = -10
			change_controller = ZOM
			secede_province = ZOM
		}
		ZOM = {
			remove_country_modifier = do_not_annex
			remove_country_modifier = accepts_provinces
			add_country_modifier = {
				name = do_not_annex
				duration = 30
			}
			country_event = { id = 123456202 days = 31 }
		}
	}
}
