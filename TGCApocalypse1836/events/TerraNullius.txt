province_event = {   #Zombify province
	id = 66100
	title = "EVTNAME66025"
	desc = "EVTDESC66025"
	picture = "prov_plague"

	trigger = {
		OR = {
			AND = {
				NOT = { has_global_flag = plague_inc_mode }
				NOT = { owner = { tag = ZOM } }
				province_control_days = 550
				has_province_modifier = zombie_outbreak
				controlled_by = REB
				exists = ZOM
			}
			AND = {
				NOT = { has_global_flag = plague_inc_mode }
				NOT = { owner = { tag = ZOM } }
				owner = { ai = yes }
				province_control_days = 225
				has_province_modifier = zombie_outbreak
				controlled_by = REB
				exists = ZOM
			}
			AND = {
				has_global_flag = plague_inc_mode 
				NOT = { owner = { tag = ZOM } }
				province_control_days = 30
				controlled_by = ZOM
				exists = ZOM
			}
		}
	}

	mean_time_to_happen = {
		months = 1
		modifier = { 
			factor = 0.1 
			OR = { 
				has_global_flag = plague_inc_mode 
				has_empty_adjacent_province = yes
				any_neighbor_province = { owner = { tag = ZOM } } 
				owner = { ai = yes }
			}
		}
	}

	option = {
		name = "DAMN"
		owner = { prestige = -3 plurality = -5 }
		JAN = {
			change_variable = {
					which = zombified_provinces
					value = 1
			}
			add_country_modifier = { name = ap_zm_timer duration = 365 }
		}
		any_pop = {
			reduce_pop = 0.8
			militancy = 8
			consciousness = -3
		}
		change_controller = ZOM
		secede_province = ZOM
		ZOM = { money = 5000 }
		add_province_modifier = { name = ap_prov_timer duration = 30 }
	}
}

#Note: this must be a country event, otherwise it might happen multiple times at once, which can cause a crash
country_event = { 
	id = 66101
	title = "EVTNAME66101"
	desc = "EVTDESC66101"
	picture = plague

	trigger = {
		NOT = { exists = ZOM }
		any_owned_province = {
			province_control_days = 365
			has_province_modifier = zombie_outbreak
			controlled_by = REB
		}
	}

	mean_time_to_happen = {
		days = 30
	}

	allow_multiple_instances = no

	option = {
		name = "DAMN"
		random_owned = {
			limit = {
				province_control_days = 365
				has_province_modifier = zombie_outbreak
				controlled_by = REB
			}
			any_pop = {
				reduce_pop = 0.8
				militancy = 8
				consciousness = -3
			}
			owner = { prestige = -3 }
			change_controller = ZOM
			secede_province = ZOM
		}
		ZOM = { add_country_modifier = { name = wasteland duration = -1 } }
	}
}

	title = "EVTNAME66200"
	desc = "EVTDESC66200"
	picture = "prov_test"

	trigger = {
		OR = { owner = { tag = ZOM }
			  NOT = { total_pops = 30 }
		}
		exists = WAS
		province_control_days = 365
	}

	mean_time_to_happen = { days = 1 }

	option = {
		name = "Zombies deleted"
		any_pop = {
			militancy = -9 
			reduce_pop = 0.1
			secede_province = WAS
		}
	}
}
